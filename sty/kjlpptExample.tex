\begin{lstlisting}[style=rocq,caption={A Specification of \texttt{prog}},label=lst:wpprog,firstnumber=1]
Lemma wp_prog
  : {{{ True }}} prog {{{ v, RET v; $\lceil$v = #0$\rceil$ $\lor$ $\lceil$v = #1$\rceil$ }}}.
Proof.
  iIntros "%$\Phi$ _ H$\Phi$".
  rewrite /prog.
  wp_alloc l as "Hl".
  wp_pures.
  (**
    Fork does not have its own tactic. Instead, we use its specification.
    This specification forces us to split our resources between the threads.
  *)
  wp_apply (wp_fork with "[Hl]").
  - iModIntro. wp_store. iModIntro. done.
  - wp_pures. Fail wp_load.
Abort.
\end{lstlisting}
